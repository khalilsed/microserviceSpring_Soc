CREATE DATABASE csv_db;
USE csv_db;
CREATE TABLE flights (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    year INT NOT NULL,
    month INT NOT NULL,
    day INT NOT NULL,
    dep_delay INT,
    arr_delay INT,
    origin VARCHAR(255),
    dest VARCHAR(255),
    air_time INT,
    distance INT,
    hour INT,
    minute INT,
    name VARCHAR(255)
);

select * from csv_db.flights;
